#include <Wire.h> 
//Direccion I2C de la IMU

#define MPU 0x68
 
//MPU-6050 da los valores en enteros de 16 bits
//Valores sin refinar
int16_t AcX, AcY, AcZ, GyX, GyY, GyZ, T;

void setup() {
  Wire.begin();
  Wire.beginTransmission(MPU);
  Wire.write(0x6B);
  Wire.write(0);
  Wire.endTransmission(true);
  Serial.begin(9600);
}

void loop() {
   //Leer los valores del Acelerometro de la IMU
   Wire.beginTransmission(MPU);
   Wire.write(0x3B); //Pedir el registro 0x3B - correspnde a la aceleracion en x  
   Wire.endTransmission(false);
   Wire.requestFrom(MPU,12,true); //A partir del 0x3B, se piden 12 registros
   AcX=Wire.read()<<8|Wire.read(); //Cada valor ocupa 2 registros
   AcY=Wire.read()<<8|Wire.read();
   AcZ=Wire.read()<<8|Wire.read();
   T=Wire.read()<<8|Wire.read();

   Serial.print(" /AcZ: "); Serial.print(AcZ);

   Serial.print(" /GyY: "); Serial.print(GyY);
   Serial.print("\n");
   
   delay(10); //Tiempo entre muestra, 0.01
}
